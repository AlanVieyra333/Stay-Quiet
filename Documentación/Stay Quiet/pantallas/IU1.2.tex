\subsection{IU1.2 Registrar Usuario}

\subsubsection{Objetivo}
	El usuario se  registrara en la aplicación llenando el formulario.  

\subsubsection{Diseño}
	Se despliega al dar clic en el \IUbutton{REGISTRATE}.

\IUfig[.3]{IU1-2}{IU1.2}{Registrar Usuario}

\subsubsection{Entradas}
Los datos del usuario requeridos:
\begin{itemize}	
	\item Nombre.
	\item Apellidos.
	\item Correo Electrónico.
	\item Numero de Celular.
	\item Contraseña y/o PIN.
\end{itemize}

\subsubsection{Salidas}
\begin{Citemize}
	\item \MSGref{MSJ1.1}{Campos Incompletos}
	\item \MSGref{MSJ1.2}{Valor no válido}
	\item \MSGref{MSJ1.3}{Error en conexión}
	\item \MSGref{MSJ1.5}{Datos ya utilizados}
\end{Citemize}
