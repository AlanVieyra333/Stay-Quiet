\subsection{IU-07 Agregar Contacto}

\subsubsection{Objetivo}
	El usuario agregar la información necesaria para el registro del Protegido.

\subsubsection{Diseño}
	Se despliega al dar clic en el \IUbutton{Agregar contacto} de la pantalla \IUref{IU-03}{Home}.

\IUfig[.3]{IU1-6}{IU-07}{Agregar Contacto}

\subsubsection{Entradas}
Los datos del usuario requeridos:
\begin{itemize}
	\item Nombre de usuario.
\end{itemize}

\subsubsection{Salidas}
\begin{Citemize}
	\item \MSGref{MSJ-01}{Campos Incompletos}
	\item \MSGref{MSJ-06}{Error en conexión}
	\item \MSGref{MSJ-14}{Nombre de usuario no existe}
\end{Citemize}
