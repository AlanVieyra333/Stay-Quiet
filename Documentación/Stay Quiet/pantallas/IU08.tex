\subsection{IU-08 Nueva contraseña}

\subsubsection{Objetivo}
	El usuario podrá crear una nueva contraseña.
\subsubsection{Diseño}
	\begin{itemize}
		\item Se despliega al dar clic en el \IUbutton{Seguridad} de la pantalla \IUref{IU-03}{Home}.
\end{itemize}		

\IUfig[.3]{IU1-9}{IU-08}{Nueva contraseña}

\subsubsection{Entradas}
Los datos de usuario a ingresar:
\begin{itemize}
	\item Contraseña anterior.
	\item Nueva contraseña
	\item Confirmar contraseña
\end{itemize}

\subsubsection{Salidas}
\begin{Citemize}
	\item \MSGref{MSJ-01}{Campos Incompletos}
	\item \MSGref{MSJ-06}{Error en conexión}
	\item \MSGref{MSJ-09}{Contraseña incorrecta}
	\item \MSGref{MSJ-10}{Formato de correo incorrecto}
\end{Citemize}
