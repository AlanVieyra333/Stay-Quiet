\subsection{IU1.8 Olvidó contraseña}

\subsubsection{Objetivo}
	El usuario introducirá el correo electrónico donde se le enviara la contraseña.
\subsubsection{Diseño}
	\begin{itemize}
		\item Se despliega al dar clic en el \IUbutton{OLVIDASTE LA CONTRASEÑA} de la pantalla \IUref{IU1.1}{Pantalla Principal}.
\end{itemize}		

\IUfig[.3]{IU1-8}{IU1.8}{Olvidó contraseña}

\subsubsection{Entradas}
Los datos de usuario a ingresar:
\begin{itemize}
	\item Correo electrónico.
\end{itemize}

\subsubsection{Salidas}
\begin{Citemize}
	\item \MSGref{MSJ1.1}{Campos Incompletos}
	\item \MSGref{MSJ1.10}{Formato de correo incorrecto}
	\item \MSGref{MSJ1.6}{Error en conexión}
\end{Citemize}
