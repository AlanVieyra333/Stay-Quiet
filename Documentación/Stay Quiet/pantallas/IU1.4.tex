\subsection{IU1.4 Perfil Usuario}

\subsubsection{Objetivo}
	El usuario vera sus datos y los podrá modificar.  

\subsubsection{Diseño}
	Se despliega al dar clic en el \IUbutton{Perfil} del menú superior derecho.

\IUfig[.3]{IU1-4}{IU1.4}{Perfil Usuario}

\subsubsection{Entradas}
Los datos de usuario a modificar:
\begin{itemize}	
	\item Nombre.
	\item Apellidos.
	\item Correo Electrónico.
	\item Numero de Celular.
\end{itemize}

\subsubsection{Salidas}
\begin{Citemize}
	\item \MSGref{MSJ1.1}{Campos Incompletos}
	\item \MSGref{MSJ1.2}{Formato de correo incorrecto}
	\item \MSGref{MSJ1.3}{Error en conexión}
\end{Citemize}
