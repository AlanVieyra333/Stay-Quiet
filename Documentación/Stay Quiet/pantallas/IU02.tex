\subsection{IU-02 Registrar}

\subsubsection{Objetivo}
	El usuario se  registrara en la aplicación llenando el formulario.

\subsubsection{Diseño}
	Se despliega al dar clic en el \IUbutton{REGISTRATE} de la pantalla \IUref{IU-01}{Iniciar Sesión}..

\IUfig[.3]{UI2}{IU-02}{Registrar}

\subsubsection{Entradas}
Los datos del usuario requeridos:
\begin{itemize}
	\item Nombre completo.
	\item Correo Electrónico.
	\item Numero de Celular.
	\item Contraseña.
\end{itemize}

\subsubsection{Salidas}
\begin{Citemize}
	\item \MSGref{MSJ-01}{Campos Incompletos}
	\item \MSGref{MSJ-02}{Formato de nombre incorrecto}
	\item \MSGref{MSJ-03}{Formato de teléfono incorrecto}
	\item \MSGref{MSJ-04}{Formato de contraseña incorrecto}
	\item \MSGref{MSJ-05}{Contraseña no coincide}
	\item \MSGref{MSJ-06}{Error en conexión}
	\item \MSGref{MSJ-08}{Datos utilizados}
	\item \MSGref{MSJ-10}{Formato de correo incorrecto}
\end{Citemize}
