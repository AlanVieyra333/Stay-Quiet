\subsection{IU1.8 Nueva contraseña}

\subsubsection{Objetivo}
	El usuario podrá crear una nueva contraseña.
\subsubsection{Diseño}
	\begin{itemize}
		\item Se despliega al dar clic en el \IUbutton{Seguridad} de la pantalla \IUref{IU1.3}{Pantalla Inicial}.
\end{itemize}		

\IUfig[.3]{IU1-9}{IU1.9}{Nueva contraseña}

\subsubsection{Entradas}
Los datos de usuario a ingresar:
\begin{itemize}
	\item Contraseña anterior.
	\item Nueva contraseña
	\item Confirmar contraseña
\end{itemize}

\subsubsection{Salidas}
\begin{Citemize}
	\item \MSGref{MSJ1.1}{Campos Incompletos}
	\item \MSGref{MSJ1.6}{Error en conexión}
	\item \MSGref{MSJ1.9}{Contraseña incorrecta}
	\item \MSGref{MSJ1.10}{Formato de correo incorrecto}
\end{Citemize}
