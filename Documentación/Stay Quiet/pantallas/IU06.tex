\subsection{IU-06 Olvidó contraseña}

\subsubsection{Objetivo}
	El usuario introducirá el correo electrónico donde se le enviara la contraseña.
\subsubsection{Diseño}
	\begin{itemize}
		\item Se despliega al dar clic en el \IUbutton{OLVIDASTE LA CONTRASEÑA} de la pantalla \IUref{IU-01}{Iniciar Sesión}.
\end{itemize}		

\IUfig[.3]{UI6}{IU-06}{Olvidó contraseña}

\subsubsection{Entradas}
Los datos de usuario a ingresar:
\begin{itemize}
	\item Correo electrónico.
\end{itemize}

\subsubsection{Salidas}
\begin{Citemize}
	\item \MSGref{MSJ-01}{Campos Incompletos}
	\item \MSGref{MSJ-10}{Formato de correo incorrecto}
	\item \MSGref{MSJ-06}{Error en conexión}
\end{Citemize}
