\begin{UseCase}{CU-02}{Registrar}{
	Se describe el comportamiento y funcionamiento de la aplicación para el registro de un usuario.
}
		\UCitem{Actor}{Usuario}
		\UCitem{Propósito}{
		Registrar a los nuevos usuarios que quieran hacer uso de la aplicación móvil.
		}
		\UCitem{Entradas}{
			Se ingresa la información:
			\begin{itemize}
				\item Nombre completo.
				\item Nombre de usuario
				\item Correo Electrónico.
				\item Numero de teléfono.
				\item Contraseña.
			\end{itemize}
		 }
		\UCitem{Salidas}{
			\begin{itemize}
				\item \MSGref{MSJ-01}{Campos Incompletos}
				\item \MSGref{MSJ-02}{Formato de nombre incorrecto}
				\item \MSGref{MSJ-03}{Formato de teléfono incorrecto}
				\item \MSGref{MSJ-04}{Formato de contraseña incorrecto}
				\item \MSGref{MSJ-06}{Error en conexión}
				\item \MSGref{MSJ-08}{Datos utilizados}
				\item \MSGref{MSJ-10}{Formato de correo incorrecto}
				\item \MSGref{MSJ-13}{Nombre de usuario existente}
				\item \MSGref{MSJ-15}{Formato de nombre de usuario incorrecto}
			\end{itemize}
		}
		\UCitem{Precondiciones}{
			No estar registrado en la aplicación.
		}
		\UCitem{Postcondiciones}{
			El usuario podrá hacer uso de la aplicación.
		}
		\UCitem{Autor}{
			\begin{itemize}
				\item Salas Hernández Abiran Natanael
			\end{itemize}
		}
	\end{UseCase}
	%-------------------------------------- COMIENZA descripción Trayectoria Crear
	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] Pulsar en el icono de la aplicación.
		\UCpaso[\UCsist] Despliega la pantalla \IUref{IU-01}{Iniciar Sesión}.
		\UCpaso[\UCactor] Pulsar en el botón \IUbutton{REGISTRATE} de la pantalla.
		\UCpaso[\UCsist] Despliega la pantalla \IUref{IU-02}{Registrar}.
		\UCpaso[\UCactor] Captura los datos necesarios en la misma pantalla.
		\UCpaso[\UCactor] Pulsar el botón \IUbutton{REGISTRAR} de la pantalla.
		\UCpaso[\UCsist] Verifica la regla de negocio \BRref{RN-01}{Campos nulos}. \Trayref{A}
		\UCpaso[\UCsist] Verifica la regla de negocio \BRref{RN-02}{Formato general de registro}. \Trayref{B}
		\UCpaso[\UCsist] Verifica la regla de negocio \BRref{RN-05}{Nombre de usuario único}. \Trayref{C}\Trayref{E}			
		\UCpaso[\UCsist] Verifica la regla de negocio \BRref{RN-03}{Usuario registrado}. \Trayref{D} 
		\UCpaso[\UCsist] Muestra la pantalla \IUref{IU-03}{Home}.
	\end{UCtrayectoria}
	%-------------------------------------Trayectoras alternativas
	\begin{UCtrayectoriaA}{A}{Condición: Faltan datos en el formulario.}
		\UCpaso[\UCsist] Muestra mensaje \MSGref{MSJ-01}{Campos Incompletos}
		\UCpaso[\UCsist] Continúa trayectoria en el paso anterior de la trayectoria principal que invocó esta trayectoria alternativa. \Trayref{Principal}
	\end{UCtrayectoriaA}
	%-------------------------------------Trayectoras alternativas
	\begin{UCtrayectoriaA}{B}{Condición: El actor captura la información de manera incorrecta.}
		\UCpaso[\UCsist] Dependiendo del error sera el mensaje:
			\begin{itemize}
			\item \MSGref{MSJ-02}{Formato de nombre incorrecto}
			\item \MSGref{MSJ-03}{Formato de teléfono incorrecto}
			\item \MSGref{MSJ-04}{Formato de contraseña incorrecto}
			\item \MSGref{MSJ-05}{Contraseña no coincide}
			\item \MSGref{MSJ-10}{Formato de correo incorrecto}
			\item \MSGref{MSJ-15}{Formato de nombre de usuario incorrecto}			
			\end{itemize}
		\UCpaso[\UCsist] Continúa trayectoria en el paso anterior de la trayectoria principal que invocó esta trayectoria alternativa. \Trayref{Principal}
	\end{UCtrayectoriaA}
	%-------------------------------------Trayectoras alternativas
	\begin{UCtrayectoriaA}{C}{Condición: Falló la conexión en la base de datos}
		\UCpaso[\UCsist] El sistema mostrará el mensaje \MSGref{MSJ-06}{Error en conexión}
		\UCpaso[\UCsist] Continúa trayectoria en el paso anterior de la trayectoria principal que invocó esta trayectoria alternativa. \Trayref{Principal}
	\end{UCtrayectoriaA}
	%-------------------------------------Trayectoras alternativas
	\begin{UCtrayectoriaA}{D}{Condición: Usuario ya registrado y/o correo y/o teléfono ya utilizados}
			\UCpaso[\UCsist] El sistema mostrará el mensaje \MSGref{MSJ-08}{Datos utilizados}
			\UCpaso[\UCsist] Continúa trayectoria en el paso 4 de la trayectoria  principal que invocó esta trayectoria alternativa. \Trayref{Principal}
	\end{UCtrayectoriaA}
	%-------------------------------------Trayectoras alternativas
	\begin{UCtrayectoriaA}{E}{Condición: Nombre de usuario ya registrado}
			\UCpaso[\UCsist] El sistema mostrará el mensaje \MSGref{MSJ-13}{Nombre de usuario existente}
			\UCpaso[\UCsist] Continúa trayectoria en el paso 4 de la trayectoria  principal que invocó esta trayectoria alternativa. \Trayref{Principal}
	\end{UCtrayectoriaA}
	%-------------------------------------Trayectoras alternativas
